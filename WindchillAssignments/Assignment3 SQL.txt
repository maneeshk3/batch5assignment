--QUERY TO GET THE MASTER DETAILS OF THE PART
select * from wtpartmaster;

--QUERY TO GET THE VERSION,HISTORY DETAILS ABOUT THE PART.
select * from wtpart;

--QUERY TO GET THE MASTER DETAILS OF PART WHOES NAME STARTS WITH PEN
select * from wtpartmaster where name like 'Pen%';

--QUERY TO GET THE HISTORY,VERSION DETAILS ABOUT ONLY 1 PARTICULAR PART WITH THE HELP OF MASTER ID
select * from wtpart JOIN  WHERE ida3masterreference = '132088';

--QUERY TO GET THE COMPLETE DETAILS FROM BOTH MASTER AND PART TABLE FOR PARTICULAR PART BASED ON ITS PRIMARY KEY.
select wtpart.*, wtpartmaster.* from wtpartmaster 
join wtpart on wtpartmaster.ida2a2=wtpart.ida3masterreference 
where wtpartmaster.ida2a2='132088';


--QUERY TO FIND ALL THE part objects with columns having partName, partNumber, Revision,Iteration,LifeCycle State
select wtpartmaster.name, wtpartmaster.wtpartnumber,wtpart.versionida2versioninfo,wtpart.iterationida2iterationinfo ,wtpart.statestate from wtpartmaster 
join  wtpart on wtpartmaster.ida2a2=wtpart.ida3masterreference 
where wtpartmaster.ida2a2='132146' order by wtpart.versionida2versioninfo,wtpart.iterationida2iterationinfo;
